def bear_room()
        puts "There is a bear here."
        puts "The bear has a bunch of honey."
        puts "The fat bear is in front of another door."
        puts "How are you going to move the bear?"
        bear_moved = false

        while true
                #kind of confused about this while loop. I'll need time to absorb it.
                prompt; next_move = gets.chomp

                if next_move == "take honey"
                        dead("The bear looks at you then slaps your face off.")
                elsif next_move == "taunt bear" and not bear_moved
                        puts "The bear has moved from the door. You can go through it now."
                        bear_moved = true
                elsif next_move == "taunt bear" and bear_moved
                        dead("The bear gets pissed off and chews your leg off.")
                elsif next_move == "open door" and bear_moved
                        gold_room()
                else
                        puts "I got no idea what that means."
                end
        end
end

def gold_room()
        puts "This room is full of gold. How much do you take?"
        # here we are getting the user's answer to the gold question
        prompt; next_move = gets.chomp
        if next_move.include?("0") or next_move.include?("1")
                how_much = next_move.to_i()
                #taking their response and turning it into an integer
        else
                dead("Man, you should have learned to type a number.")
        end

        if how_much < 50
                puts "Nice, you're not greedy, you win."
                Process.exit(0)
                #exiting the game. I like this idea of Process.exit. I'll use it again
        else
                dead("You greedy bastard!")
        end
end
